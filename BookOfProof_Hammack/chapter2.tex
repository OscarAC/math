\subsection{Statements}

\begin{exercise}{}
    Decide whether or not the following are statements. In the case of a statement, say if it is true or false, if possible.
\end{exercise}

\begin{problems}
    \problem Every even integer is a real number.
    \answer  Statement, true.

    \problem Sets $\mathbb{Z}$ and $\mathbb{N}$.
    \answer  Not a statement.

    \problem Some sets are finite.
    \answer  Statement, true.

    \problem $\mathbb{N} \notin \mathcal{P}(\mathbb{N})$
    \answer  Statement, false.

    \problem $(\mathbb{R} \times \mathbb{N}) \cap (\mathbb{N} \times \mathbb{R}) = \mathbb{N} \times \mathbb{N}$.
    \answer  Statement, true.

    \problem If the integer $x$ is a multiple of 7, then it is divisible by 7.
    \answer  Statement, true.

    \problem Call me Ishmael.
    \answer  Not a statement.
\end{problems}

\subsection{And, Or, Not}

\begin{exercise}{}
    Express each statement or open sentence in one of the forms $P \land Q$, $P \lor Q$, or $\lnot P$. Be sure to also state exactly what statements $P$ and $Q$ stand for.
\end{exercise}

\begin{problems}
    \problem The matrix $A$ is not invertible.
    \answer  $\lnot P$, where $P$: Matrix $A$ is invertible.

    \problem $x < y$.
    \answer  $\lnot P$, where $P$: $x \geq y$.

    \problem There is a quiz scheduled for Wednesday or Friday.
    \answer  $P \lor Q$, where $P$: A quiz is scheduled for Wednesday, and $Q$: A quiz is scheduled for Friday.

    \problem At least one of the numbers $x$ and $y$ equals $0$.
    \answer  $P \lor Q$, where $P$: $x = 0$, and $Q$: $y = 0$.

    \problem $x \in A \cup B$.
    \answer  $P \lor Q$, where $P$: $x \in A$, and $Q$: $x \in B$.

    \problem Happy families are all alike, but each unhappy family is unhappy in its own way. (\textit{Leo Tolstoy}, \textit{Anna Karenina})
    \answer  $P \land Q$, where $P$: Happy families are all alike, and $Q$: Each unhappy family is unhappy in its own way.

    \problem A man should look for what is, and not for what he thinks should be. (\textit{Albert Einstein})
    \answer  $P \land \lnot Q$, where $P$: Man should look for what is, and $Q$: Man should look for what he thinks should be.
\end{problems}

\subsection{Conditional Statements}

\begin{exercise}{}
    Without changing their meanings, convert each of the following sentences into a sentence having the form ``\textit{If P, then Q}''.
\end{exercise}

\begin{problems}
    \problem For a function to be continuous, it is sufficient that it is differentiable.
    \answer  If a function is differentiable, then the function is continuous.

    \problem A function is rational if it is a polynomial.
    \answer If a function is a polynomial, then the function is rational.

    \problem Whenever a surface has only one side, it is non-orientable.
    \answer If a surface has only one side, then the surface is non-orientable.

    \problem A geometric series with ratio $r$ converges if $\abs{r} < 1$.
    \answer If a geometric series has a ratio with an absolute value less than 1, then the geometric series converges.

    \problem The discriminant is negative only if the quadratic equation has no real solutions.
    \answer If a quadratic equation has a discriminant that is negative, then the quadratic equation has no real solutions.

    \problem People will generally accept facts as truth only if the facts agree with what they already believe. (\textit{Andy Rooney})
    \answer If people accept facts as truth, then the facts agree with what they already believe.
\end{problems}

\subsection{Biconditional Statements}

\begin{exercise}{}
    Without changing their meanings, convert each of the following sentences into a sentence having the form ``\textit{P if and only if Q}''.
\end{exercise}

\begin{problems}
    \problem If a function has a constant derivative then it is linear, and conversely.
    \answer A function is linear if and only if the function has a constant derivative.

    \problem If $a \in \mathbb{Q}$ then $5a \in \mathbb{Q}$, and if $5a \in \mathbb{Q}$ then $a \in \mathbb{Q}$.
    \answer $5a \in \mathbb{Q}$ if and only if $a \in \mathbb{Q}$
\end{problems}

\subsection{Quantifiers}

\begin{exercise}{}
    Write the following as English sentences, Say whether they are true or false.
\end{exercise}

\begin{problems}
    \problem $\forall x \in \mathbb{R}, \exists n \in \mathbb{N}, x^n \geq 0$
    \answer For every real number $x$, there is a natural number $n$ for which $x^n \geq 0$.\\ 
    The statement is true, consider $n = 2$.

    \problem $\forall X \in \mathcal{P}(\mathbb{N}), X \subseteq \mathbb{R}$
    \answer Every subset of the natural numbers is also a subset of the real numbers.\\
    The statement is true, since $\mathbb{N} \subseteq \mathbb{R}$.

    \problem $\exists n \in \mathbb{N}, \forall X \in \mathcal{P}(\mathbb{N}), |X| < n$
    \answer There exists a natural number $n$ for which $|X| < n$ for all subsets $X$ of the natural numbers.\\ 
    This statement is false, since $\mathbb{N} \in \mathcal{P}(\mathbb{N})$ and $\mathbb{N}$ is infinite.

    \problem $\forall n \in \mathbb{Z}, \exists X \subseteq \mathbb{N}, |X| = n$
    \answer For every integer $n$, there exists a subset $X$ of the natural numbers for which $|X| = n$.\\ 
    This statement is false, consider $n = -1$.

    \problem $\exists m \in \mathbb{Z}, \forall n \in \mathbb{Z}, m = n + 5$
    \answer There exists an integer $m$ such that for every integer $n$, $m = n + 5$.\\ 
    This statement is false, consider $n = 0$ and $n = 1$.
\end{problems}
