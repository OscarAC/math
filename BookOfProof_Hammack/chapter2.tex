\section{Logic}

\subsection{Statements}

\begin{exercise}{}
    Decide whether or not the following are statements. In the case of a statement, say if it is true or false, if possible.
\end{exercise}

\begin{solution}
    \begin{enumerate}[label=]
        \item{2.} Every even integer is a real number. $\rightarrow$ Statement, true.
        \item{4.} Sets $\mathbb{Z}$ and $\mathbb{N}$. $\rightarrow$ Not a statement.
        \item{6.} Some sets are finite. $\rightarrow$ Statement, true.
        \item{8.} $\mathbb{N} \notin \mathcal{P}(\mathbb{N}) \rightarrow$ Statement, false.
        \item{10.} $(\mathbb{R} \times \mathbb{N}) \cap (\mathbb{N} \times \mathbb{R}) = \mathbb{N} \times \mathbb{N}$. $\rightarrow$ Statement, true.
        \item{12.} If the integer $x$ is a multiple of 7, then it is divisible by 7. $\rightarrow$ Statement, true.
        \item{14.} Call me Ishmael. $\rightarrow$ Not a statement.
    \end{enumerate}
\end{solution}

\subsection{And, Or, Not}

\begin{exercise}{}
    Express each statement or open sentence in one of the forms $P \land Q$, $P \lor Q$, or $\lnot P$. Be sure to also state exactly what statements $P$ and $Q$ stand for.
\end{exercise}

\begin{solution}
    \begin{enumerate}[label=]
        \item{2.} The matrix $A$ is not invertible. $\rightarrow$ $\lnot P$, where $P:$ Matrix $A$ is invertible.
        \item{4.} $x < y$. $\rightarrow$ $\lnot P$, where $P$: $x \geq y$ 
        \item{6.} There is a quiz scheduled for Wednesday or Friday. $\rightarrow$ $P \lor Q$, where $P$: A quiz is scheduled for Wednesday, and $Q$: A quiz is scheduled for Friday.
        \item{8.} At least one of the numbers $x$ and $y$ equals $0$. $\rightarrow$ $P \lor Q$, where $P$: $x$ = 0, and $Q$: $y$ = 0.
        \item{10.} $x \in A \cup B$. $\rightarrow$ $P \lor Q$, where $P$: $x \in A$, and $Q$: $x \in B$.
        \item{12.} Happy families are all alike, but each unhappy family is unhappy in its own way. (\textit{Leo Tolstoy}, \textit{Anna Karenina}) $\rightarrow$ $P \land Q$, where $P$: Happy families are all alike, and $Q$: Each unhappy family is unhappy in its own way.
        \item{14.} A man should look for what is, and not for what he thinks should be. (\textit{Albert Einstein}) $\rightarrow$ $P \land \lnot Q$, where $P$: Man should look for what is, and $Q$: Man should look for what he thinks should be.
    \end{enumerate}
\end{solution}
